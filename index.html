async function loadData() { try { const res = await fetch('인명.csv'); const arrayBuffer = await res.arrayBuffer(); const text = new TextDecoder('utf-8').decode(arrayBuffer); const rows = parseCSV(text); const header = rows[0].map(h => h.replace(/["']/g, '').trim()); masterDB = rows.slice(1).map(r => { let obj = {}; header.forEach((h, i) => obj[h] = r[i] || ""); obj._imgs = []; obj._folders = new Set(['전체']); const personName = obj['이름'] ? obj['이름'].trim() : ""; if (obj['사진파일'] && personName) { // [핵심수정] 따옴표 제거 및 쉼표 분할 로직 강화 const rawPhotos = obj['사진파일'].replace(/^["']|["']$/g, ''); rawPhotos.split(',').forEach(p => { const relPath = p.trim().replace(/^["']|["']$/g, ''); // 개별 경로의 따옴표 또 제거 if(relPath) { const folder = relPath.includes('/') ? relPath.split('/')[0] : '기타'; obj._folders.add(folder); // 최종 경로 조립: img/이름/폴더/파일명 const fullPath = `img/${personName}/${relPath}`; obj._imgs.push({ fullPath: fullPath, folder: folder }); } }); } obj._currentTab = '전체'; return obj; }); runSearch(); } catch (e) { alert("인명.csv 로드 실패"); } }
